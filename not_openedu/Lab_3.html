def coeff_r(m, qty):
    lst_rcoef = []
    for k in range(m-1):
        ik_sum, i_sum1, k_sum, sqi_sum1 = 0, 0, 0, 0
        i_sum2, sqi_sum2 = 0, 0
        for i in range(m+k):
            if i < m:
                ik_sum += qty[i] * qty[i+k]
                i_sum1 += qty[i]
                k_sum += qty[i+k]
                sqi_sum1 += qty[i] ** 2
            if i >= k+1:
                i_sum2 += qty[i]
                sqi_sum2 += qty[i] ** 2
        r = (m * ik_sum - i_sum1 * k_sum) / (((m * sqi_sum1 - i_sum1 ** 2) * (m * sqi_sum2 - i_sum2 ** 2)) ** 0.5)
        lst_rcoef.append(r)
    return lst_rcoef
